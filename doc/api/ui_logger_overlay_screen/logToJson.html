Map<String, dynamic> logToJson(DevLogModel log) {
  if (log is ApiLogModel) {
    return {
      'id': log.id,
      'timestamp': log.timestamp.toIso8601String(),
      'type': log.type,
      'heading': log.heading,
      'content': log.content,
      'method': log.method,
      'url': log.url,
      'headers': log.headers,
      'body': log.body,
      'statusCode': log.statusCode,
      'timings': log.timings,
      'memoryUsage': log.memoryUsage,
    };
  } else {
    return {
      'id': log.id,
      'timestamp': log.timestamp.toIso8601String(),
      'type': log.type,
      'heading': log.heading,
      'content': log.content,
    };
  }
}
